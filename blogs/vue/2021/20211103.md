## 一、获取省市区数据源

```shell
npm i china-division
```

## 二、组件封装

```vue
<template>
  <div id="tip">
    <a-select
      v-model="provinceValue"
      placeholder="省"
      class="c-amap_select"
      @change="provinceChange"
    >
      <a-select-option
        v-for="item in provinceOptions"
        :key="item.code"
        :value="item.code"
      >
        {{ item.name }}
      </a-select-option>
    </a-select>
    <a-select
      v-model="cityValue"
      placeholder="市"
      class="c-amap_select"
      @change="cityChange"
    >
      <a-select-option
        v-for="item in cityOptions"
        :key="item.code"
        :value="item.code"
      >
        {{ item.name }}
      </a-select-option>
    </a-select>
    <a-select
      v-model="areaValue"
      placeholder="区"
      class="c-amap_select"
      @change="areaChange"
    >
      <a-select-option
        v-for="item in areaOptions"
        :key="item.code"
        :value="item.code"
      >
        {{ item.name }}
      </a-select-option>
    </a-select>
  </div>
</template>

<script>
import provinces from 'china-division/dist/provinces.json'
import cities from 'china-division/dist/cities.json'
import areas from 'china-division/dist/areas.json'
export default {
  props: {
    provinceProp: {
      type: String,
      default: ''
    },
    cityProp: {
      type: String,
      default: ''
    },
    areaProp: {
      type: String,
      default: ''
    },
  },
  data() {
    return {
      input: '',
      provinceValue: '',
      cityValue: '',
      areaValue: '',
      provinceOptions: [],
      cityOptions: [],
      areaOptions: [],
    }
  },
  mounted() {
    // console.log(provinces, cities, areas)
    this.init()
    this.loadData()
  },
  methods: {
    init() {
      if (this.provinceProp) {
        this.provinceValue = this.provinceProp
        this.provinceChange()
      }
      if (this.cityProp) {
        this.cityValue = this.cityProp
        this.cityChange()
      }
      if (this.areaProp) {
        this.areaValue = this.areaProp
        this.areaChange()
      }
    },
    loadData() {
      this.provinceOptions = provinces
    },
    resetCityAndareaData() {
      this.cityValue = ''
      this.areaValue = ''
      this.cityOptions = []
      this.areaOptions = []
    },
    resetareaData() {
      this.areaValue = ''
      this.areaOptions = []
    },
    provinceChange() {
      this.resetCityAndareaData()
      // console.log(v)
      this.cityOptions = cities.filter(city => city.provinceCode === this.provinceValue)

    },
    cityChange() {
      this.resetareaData()
      // console.log(v)
      this.areaOptions = areas.filter(area => area.cityCode === this.cityValue)
      if (!this.isArea) {
        this.chooseData()
      }
    },
    areaChange() {
      this.chooseData()
    },
    chooseData() {
      // console.log(this.provinceValue, this.cityValue, this.areaValue)
      const data = {
        provinceValue: this.provinceValue ? this.provinceValue : '',
        cityValue: this.cityValue ? this.cityValue : '',
        areaValue: this.areaValue ? this.areaValue : ''
      }
      this.$emit('change', data)
    }
  }
}
</script>
<style lang="scss">
.c-amap_select {
  width: 100px !important;
}
</style>

```
